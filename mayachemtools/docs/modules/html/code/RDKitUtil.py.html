  44 def GetMolName(Mol, MolNum = None):
  45     """Get molecule name.
  46     
  47     Arguments:
  48         Mol (object): RDKit molecule object.
  49         MolNum (int or None): Molecule number in input file.
  50 
  51     Returns:
  52         str : Molname corresponding to _Name property of a molecule, generated
  53             from specieid MolNum using the format "Mol%d" % MolNum, or an
  54             empty string.
  55 
  56     """
  57     
  58     MolName = ''
  59     if Mol.HasProp("_Name"):
  60         MolName = Mol.GetProp("_Name")
  61 
  62     if not len(MolName):
  63         if MolNum is not None:
  64             MolName = "Mol%d" % MolNum
  65     
  66     return MolName

 117 def _ModifySVGForInlineEmbedding(SVGText, Base64Encoded):
 118     """Modify SVG for inline embedding into a HTML page using "img" tag
 119     along with performing base64 encoding.
 120     """
 121     
 122     # Take out all tags till the start of '<svg' tag...
 123     Pattern = re.compile("^.*<svg", re.I | re.S)
 124     SVGText = Pattern.sub("<svg", SVGText)
 125     
 126     # Add an extra space before the "width=..." tag. Otherwise, inline embedding may
 127     # cause the following XML error on some browsers due to start of the "width=..."
 128     # at the begining of the line in <svg ...> tag:
 129     #
 130     #  XML5607: Whitespace expected.
 131     #
 132     SVGText = re.sub("width='", " width='", SVGText, flags = re.I)
 133     
 134     # Take out trailing new line...
 135     SVGText = SVGText.strip()
 136 
 137     # Perform base64 encoding by turning text into byte stream using string
 138     # encode and transform byte stream returned by b64encode into a string
 139     # by string decode...
 140     #
 141     if Base64Encoded:
 142         SVGText = base64.b64encode(SVGText.encode()).decode()
 143 
 144     return SVGText

 175 def GetSVGForMolecules(Mols, MolsPerRow, MolWidth, MolHeight, Legends = None, AtomListsToHighlight = None, BondListsToHighlight = None, BoldText = True):
 176     """Get SVG image text for molecules suitable for viewing in a browser.
 177     
 178     Arguments:
 179         Mols (list): List of RDKit molecule objects.
 180         MolsPerRow (int): Number of molecules per row.
 181         Width (int): Width of a molecule image in pixels.
 182         Height (int): Height of a molecule image in pixels.
 183         Legends (list): List containing strings to display under images.
 184         AtomListsToHighlight (list): List of lists containing atoms to highlight
 185             for molecules.
 186         BondListsToHighlight (list): List of lists containing bonds to highlight
 187             for molecules
 188         BoldText (bool): Flag to make text bold in the image of molecules. 
 189 
 190     Returns:
 191         str : SVG image text for writing to a SVG file for viewing in a browser.
 192 
 193     """
 194     
 195     SVGText = Draw.MolsToGridImage(Mols, molsPerRow = MolsPerRow, subImgSize = (MolWidth,MolHeight), legends = Legends, highlightAtomLists = AtomListsToHighlight, highlightBondLists = BondListsToHighlight, useSVG = True)
 196     
 197     return _ModifySVGForBrowserViewing(SVGText, BoldText)

 199 def _ModifySVGForBrowserViewing(SVGText, BoldText = True):
 200     """Modify SVG for loading into a browser."""
 201     
 202     # It appears that the string 'xmlns:svg' needs to be replaced with 'xmlns' in the
 203     # SVG image string generated by RDKit. Otherwise, the image doesn't load in 
 204     # web browsers.
 205     #
 206     if re.search("xmlns:svg", SVGText, re.I):
 207         SVGText = re.sub("xmlns:svg", "xmlns", SVGText, flags = re.I)
 208     
 209     # Make text bold...
 210     if BoldText:
 211         SVGText = re.sub("font-weight:normal;", "font-weight:bold;", SVGText, flags = re.I)
 212     
 213     return SVGText

 230 def AreAtomIndicesSequentiallyConnected(Mol, AtomIndices):
 231     """Check for the presence bonds between sequential pairs of atoms in a
 232     molecule.
 233     
 234     Arguments:
 235         Mol (object): RDKit molecule object.
 236         AtomIndices (list): List of atom indices.
 237 
 238     Returns:
 239         bool : True - Sequentially connected; Otherwise, false. 
 240 
 241     """
 242 
 243     for Index in range(0, (len(AtomIndices) -1)):
 244         if Mol.GetBondBetweenAtoms(AtomIndices[Index], AtomIndices[Index + 1]).GetIdx() is None:
 245             return False
 246     
 247     return True

 250     """Encode RDkit molecule object into a base64 encoded string. The properties
 251     can be optionally excluded.
 252     
 253     The molecule is pickled using RDKit Mol.ToBinary() function before
 254     their encoding.
 255    
 256     Arguments:
 257         Mol (object): RDKit molecule object.
 258         PropertyPickleFlags: RDKit property pickle options.
 259 
 260     Returns:
 261         str : Base64 encode molecule string or None.
 262 
 263     Notes:
 264         The following property pickle flags are currently available in RDKit:

 266             Chem.PropertyPickleOptions.NoProps

 337         Mol (object): RDKit molecule object.
 338         PatternMol (object): RDKit molecule object for a SMILES/SMARTS pattern.
 339         AtomIndices (list): Atom indices.
 340         FilterByAtomMapNums (bool): Filter matches by atom map numbers.
 341 
 342     Returns:
 343         object : RDKit molecule object or None.
 344 
 345     """

 351 def MolsFromSubstructureMatches(Mol, PatternMol, AtomIndicesList, FilterByAtomMapNums = False):
 352     """Generate  a list of RDKit molecule objects for a list containing lists of
 353     matched atom indices present in a pattern molecule. The list of atom indices
 354     correspond to a list retrieved by RDKit function GetSubstructureMatches using
 355     SMILES/SMARTS pattern. The atom indices are optionally filtered by mapping
 356     atom numbers to appropriate atom indices during the generation of the molecule. For
 357     example: [O:1]=[S:2](=[O])[C:3][C:4].
 358      
 359     Arguments:
 360         Mol (object): RDKit molecule object.
 361         PatternMol (object): RDKit molecule object for a SMILES/SMARTS pattern.
 362         AtomIndicesList (list): A list of lists containing atom indices.
 363         FilterByAtomMapNums (bool): Filter matches by atom map numbers.
 364 
 365     Returns:
 366         list : A list of lists containg RDKit molecule objects or None.
 367 
 368     """
 369 
 370     AtomMapIndices = _GetAtomMapIndices(PatternMol) if FilterByAtomMapNums else None

 422 def _MolFromSubstructureMatch(Mol, PatternMol, AtomIndices, AtomMapIndices):
 423     """Generate a RDKit molecule object for a list of matched atom indices and available
 424    atom map indices.
 425     """
 426 
 427     if AtomMapIndices is not None:
 428         MatchedAtomIndices = [AtomIndices[Index] for Index in AtomMapIndices]
 429     else:
 430         MatchedAtomIndices = list(AtomIndices)
 431 
 432     return _GetMolFromAtomIndices(Mol, MatchedAtomIndices)

 434 def _GetAtomMapIndices(PatternMol):
 435     """Get a list of any available atom indices corresponding to atom map numbers
 436     present in  pattern SMILES/SMARTS used for creating pattern molecule.
 437     """
 438 
 439     # Setup a atom map number to atom indices map..
 440     AtomMapNumToIndices = {}
 441     for Atom in PatternMol.GetAtoms():
 442         AtomMapNum = Atom.GetAtomMapNum()
 443         
 444         if AtomMapNum:
 445             AtomMapNumToIndices[AtomMapNum] = Atom.GetIdx()
 446     
 447     # Setup atom indices corresponding to atom map numbers...
 448     AtomMapIndices = None
 449     if len(AtomMapNumToIndices):
 450         AtomMapIndices = [AtomMapNumToIndices[AtomMapNum] for AtomMapNum in sorted(AtomMapNumToIndices)]
 451 
 452     return AtomMapIndices

 455     """Filter substructure match atom indices by atom map indices corresponding to
 456     atom map numbers.
 457     """
 458     
 459     if AtomMapIndices is None:
 460         return list(AtomIndices)

 464 def _GetMolFromAtomIndices(Mol, AtomIndices):
 465     """Generate a RDKit molecule object from atom indices returned by
 466    substructure search.
 467     """
 468 
 469     BondIndices = []
 470     for AtomIndex in AtomIndices:
 471         Atom = Mol.GetAtomWithIdx(AtomIndex)
 472         
 473         for AtomNbr in Atom.GetNeighbors():
 474             AtomNbrIndex = AtomNbr.GetIdx()
 475             if AtomNbrIndex not in AtomIndices:
 476                 continue
 477             
 478             BondIndex = Mol.GetBondBetweenAtoms(AtomIndex, AtomNbrIndex).GetIdx()
 479             if BondIndex in BondIndices:
 480                 continue
 481                 
 482             BondIndices.append(BondIndex)
 483             
 484     MatchedMol = Chem.PathToSubmol(Mol, BondIndices) if len(BondIndices) else None
 485     
 486     return MatchedMol

 489 def ReadAndValidateMolecules(FileName, **KeyWordArgs):
 490     """Read molecules from an input file, validate all molecule objects, and return
 491     a list of valid and non-valid molecule objects along with their counts.
 492     
 493     Arguments:
 494         FileName (str): Name of a file with complete path.
 495         **KeyWordArgs (dictionary) : Parameter name and value pairs for reading and
 496             processing molecules.
 497 
 498     Returns:
 499         list : List of valid RDKit molecule objects.
 500         int : Number of total molecules in input file. 
 501         int : Number of valid molecules in input file. 
 502 
 503     Notes:
 504         The file extension is used to determine type of the file and set up an appropriate
 505         file reader.
 506 
 507     """
 508 
 509     AllowEmptyMols = True
 510     if "AllowEmptyMols" in KeyWordArgs:
 511         AllowEmptyMols = KeyWordArgs["AllowEmptyMols"]
 512     
 513     Mols = ReadMolecules(FileName, **KeyWordArgs)
 514 
 515     if AllowEmptyMols:
 516         ValidMols = [Mol for Mol in Mols if Mol is not None]
 517     else:
 518         ValidMols = []
 519         MolCount = 0
 520         for Mol in Mols:
 521             MolCount += 1
 522             if Mol is None:
 523                 continue
 524             
 525             if IsMolEmpty(Mol):
 526                 MolName = GetMolName(Mol, MolCount)
 527                 MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 528                 continue
 529             
 530             ValidMols.append(Mol)
 531             
 532     MolCount = len(Mols)
 533     ValidMolCount = len(ValidMols)
 534 
 535     return (ValidMols, MolCount, ValidMolCount)

 537 def ReadMolecules(FileName, **KeyWordArgs):
 538     """Read molecules from an input file without performing any validation
 539     and creation of molecule objects.
 540     
 541     Arguments:
 542         FileName (str): Name of a file with complete path.
 543         **KeyWordArgs (dictionary) : Parameter name and value pairs for reading and
 544             processing molecules.
 545 
 546     Returns:
 547         list : List of RDKit molecule objects.
 548 
 549     Notes:
 550         The file extension is used to determine type of the file and set up an appropriate
 551         file reader.
 552 
 553     """
 554 
 555     # Set default values for possible arguments...
 556     ReaderArgs = {"Sanitize": True, "RemoveHydrogens": True, "StrictParsing": True,  "SMILESDelimiter" : ' ', "SMILESColumn": 1, "SMILESNameColumn": 2, "SMILESTitleLine": True }
 557 
 558     # Set specified values for possible arguments...
 559     for Arg in ReaderArgs:
 560         if Arg in KeyWordArgs:
 561             ReaderArgs[Arg] = KeyWordArgs[Arg]
 562 
 563     # Modify specific valeus for SMILES...
 564     if MiscUtil.CheckFileExt(FileName, "smi csv tsv txt"):
 565         Args = ["Sanitize", "SMILESTitleLine"]
 566         for Arg in Args:
 567             if ReaderArgs[Arg] is True:
 568                 ReaderArgs[Arg] = 1
 569             else:
 570                 ReaderArgs[Arg] = 0
 571     
 572     Mols = []
 573     if MiscUtil.CheckFileExt(FileName, "sdf sd"):
 574         return ReadMoleculesFromSDFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"], ReaderArgs['StrictParsing'])
 575     elif MiscUtil.CheckFileExt(FileName, "mol"):
 576         return ReadMoleculesFromMolFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"], ReaderArgs['StrictParsing'])
 577     elif MiscUtil.CheckFileExt(FileName, "mol2"):
 578         return ReadMoleculesFromMol2File(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"])
 579     elif MiscUtil.CheckFileExt(FileName, "pdb"):
 580         return ReadMoleculesFromPDBFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"])
 581     elif MiscUtil.CheckFileExt(FileName, "smi txt csv tsv"):
 582         SMILESColumnIndex = ReaderArgs["SMILESColumn"] - 1
 583         SMILESNameColumnIndex = ReaderArgs["SMILESNameColumn"] - 1
 584         return ReadMoleculesFromSMILESFile(FileName, ReaderArgs["SMILESDelimiter"], SMILESColumnIndex, SMILESNameColumnIndex, ReaderArgs["SMILESTitleLine"], ReaderArgs["Sanitize"])
 585     else:
 586         MiscUtil.PrintWarning("RDKitUtil.ReadMolecules: Non supported file type: %s" % FileName)
 587     
 588     return Mols

 605 def ReadMoleculesFromMolFile(FileName, Sanitize = True, RemoveHydrogens = True, StrictParsing = True):

 674 def MoleculesWriter(FileName, **KeyWordArgs):
 675     """Set up a molecule writer.
 676     
 677     Arguments:
 678         FileName (str): Name of a file with complete path.
 679         **KeyWordArgs (dictionary) : Parameter name and value pairs for writing and
 680             processing molecules.
 681 
 682     Returns:
 683         RDKit object : Molecule writer.
 684 
 685     Notes:
 686         The file extension is used to determine type of the file and set up an appropriate
 687         file writer.
 688 
 689     """
 690     
 691     # Set default values for possible arguments...
 692     WriterArgs = {"Compute2DCoords" : False, "Kekulize": False, "SMILESDelimiter" : ' ', "SMILESIsomeric": True, "SMILESTitleLine": True, "SMILESMolName": True}
 693 
 694     # Set specified values for possible arguments...
 695     for Arg in WriterArgs:
 696         if Arg in KeyWordArgs:
 697             WriterArgs[Arg] = KeyWordArgs[Arg]
 698     
 699     Writer = None
 700     if MiscUtil.CheckFileExt(FileName, "sdf sd"):
 701         Writer = Chem.SDWriter(FileName)
 702         if WriterArgs["Kekulize"]:
 703             Writer.SetKekulize(True)
 704     elif MiscUtil.CheckFileExt(FileName, "pdb"):
 705         Writer = Chem.PDBWriter(FileName)
 706     elif MiscUtil.CheckFileExt(FileName, "smi"):
 707         # Text for the name column in the title line. Blank indicates not to include name column
 708         # in the output file...
 709         NameHeader = 'Name' if WriterArgs["SMILESMolName"] else ''
 710         Writer = Chem.SmilesWriter(FileName, delimiter = WriterArgs["SMILESDelimiter"], nameHeader = NameHeader, includeHeader = WriterArgs["SMILESTitleLine"],  isomericSmiles = WriterArgs["SMILESIsomeric"], kekuleSmiles = WriterArgs["Kekulize"])
 711     else:
 712         MiscUtil.PrintWarning("RDKitUtil.WriteMolecules: Non supported file type: %s" % FileName)
 713     
 714     return Writer
 715     
 716 def WriteMolecules(FileName, Mols, **KeyWordArgs):
 717     """Write molecules to an output file.
 718     
 719     Arguments:
 720         FileName (str): Name of a file with complete path.
 721         Mols (list): List of RDKit molecule objects. 
 722         **KeyWordArgs (dictionary) : Parameter name and value pairs for writing and
 723             processing molecules.
 724 
 725     Returns:
 726         int : Number of total molecules.
 727         int : Number of processed molecules written to output file.
 728 
 729     Notes:
 730         The file extension is used to determine type of the file and set up an appropriate
 731         file writer.
 732 
 733     """
 734     
 735     Compute2DCoords = False
 736     if "Compute2DCoords" in KeyWordArgs:
 737         Compute2DCoords = KeyWordArgs["Compute2DCoords"]
 738     
 739     SetSMILESMolProps = KeyWordArgs["SetSMILESMolProps"] if "SetSMILESMolProps" in KeyWordArgs else False
 740         
 741     MolCount = len(Mols)
 742     ProcessedMolCount = 0
 743     
 744     Writer = MoleculesWriter(FileName, **KeyWordArgs)
 745     
 746     if Writer is None:
 747         return (MolCount, ProcessedMolCount)
 748     
 749     FirstMol = True
 750     for Mol in Mols:
 751         if Mol is None:
 752             continue
 753 
 754         if FirstMol:
 755             FirstMol = False
 756             if SetSMILESMolProps:
 757                 SetWriterMolProps(Writer, Mol)
 758                 
 759         ProcessedMolCount += 1
 760         if Compute2DCoords:
 761             AllChem.Compute2DCoords(Mol)
 762         
 763         Writer.write(Mol)
 764     
 765     Writer.close()
 766     
 767     return (MolCount, ProcessedMolCount)

 769 def SetWriterMolProps(Writer, Mol):
 770     """Setup molecule properties for a writer to output.
 771     
 772     Arguments:
 773         Writer (object): RDKit writer object.
 774         Mol (object): RDKit molecule object.
 775 
 776     Returns:
 777         object : Writer object.
 778 
 779     """
 780     PropNames = list(Mol.GetPropNames())
 781     if len(PropNames):
 782         Writer.SetProps(PropNames)
 783         
 784     return Writer
